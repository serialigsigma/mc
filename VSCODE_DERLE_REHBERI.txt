#!/bin/bash

# Light Client Kurulum Script'i - Linux/Mac

echo "==================================="
echo "Light Client Fabric Installer"
echo "==================================="
echo ""

if [ ! -f "gradlew" ]; then
    echo "HATA: gradlew bulunamadi!"
    echo "Dosyayi fabric-client dizininde calistir."
    exit 1
fi

echo "[1/3] Gradle wrapper indiriliyor..."
chmod +x gradlew
./gradlew wrapper > /dev/null 2>&1

echo "[2/3] Mod derleniyor..."
./gradlew build

echo "[3/3] Tamamlandi!"
echo ""
echo "JAR dosyasi olusturuldu: build/libs/lightclient-1.0.0.jar"
echo ""
echo "SONRAKI ADIM:"
echo "1. JAR dosyasini:"
echo "   - Windows: %APPDATA%\.minecraft\mods\"
echo "   - Mac: ~/Library/Application Support/minecraft/mods/"
echo "   - Linux: ~/.minecraft/mods/"
echo "   klasorune kopyala"
echo "2. Minecraft'i ac ve Fabric profile'ini sec"
echo "3. Oyuna gir!"
echo ""
